Sai.on() events.

(function(window){

  window.Sai && Sai.server = "//log.example.com/sai.gif";

  seajs.use("sai/3.0.0/sai", function(Sai){

    Sai.on("*", function(meta){
      meta.uid = "user_id";
    });

    Sai.on("jserror", function(meta){
      meta.error_id = "error_id";
    });

  });

})(this);
